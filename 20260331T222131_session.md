>   static decompressLZF(input: Uint8Array, startOffset: number, output: Uint8Array, outputOffset: number, maxBytes: number): number[] {
>     let i = startOffset;
>     let j = outputOffset;
>     const outputLimit = outputOffset + maxBytes;
>     
>     while (i < input.length && j < outputLimit) {
>       const ctrl = input[i++];
> 
>       if (ctrl < 32) {
>         // Literal run
>         let len = ctrl + 1;
>         if (j + len > outputLimit) len = outputLimit - j;
>         for (let k = 0; k < len; k++) output[j++] = input[i++];
>       } else {
>         // Back-reference
>         let len = ctrl >> 5;
>         let offset = ((ctrl & 0x1f) << 8);
>         if (i < input.length) offset += input[i++];

>         if (len === 7 && i < input.length) {

>         }

>               output[j] = output[start + k];

>           }
>         }
>       }
>     }

> ### 1. `krita-tiles.ts` Analizi (Hata Kaynakları)
> Kodunuzdaki `parse` metodu şu an oldukça spesifik varsayımlara dayanıyor ve Krita'nın farklı sürümleri/dosya yapıları arasında değişkenlik gösteren "Binary Tiled Format"ı tam karşılamıyor olabilir:
> 
> *   **Delta Encoding:** Kodda `Undo Planar Horizontal Delta` yapılıyor: `tileData[planeStart + i] = (tileData[planeStart + i] + tileData[planeStart + i - 1]) & 0xFF;`. Ancak Krita bazen **dikey delta** veya **hiç delta olmayan** (sadece planar) veri kullanabiliyor. Testlerinizdeki `test_planar_skip1_nodelta.png` gibi dosya isimleri, bunun bir değişken olduğunu zaten gösteriyor.
> *   **Planar Sıralaması:** `pixels[targetIdx + 0] = tileData[srcIdx + 2 * planeSize];` gibi doğrudan indeksleme, Krita'nın dosya içindeki renk kanalı sırasını (B, G, R, A vs. R, G, B, A) sabit bir varsayımla okuyor.
> *   **Decompression Skip:** `if (dLen > expectedSize) { skip = dLen - expectedSize; }` satırı çok riskli. Krita'nın tile başlık bilgisi (bazı versiyonlarda 16-48 byte arası) değişken olabilir. Sabit `skip` hesaplaması veriyi kaydırıyor ve sonuç görüntünün "scrambled" (karışık) görünmesine sebep oluyor.
>

>     let i = startOffset;
>     let j = outputOffset;
>     const outputLimit = outputOffset + maxBytes;
>     
>     while (i < input.length && j < outputLimit) {
>       const ctrl = input[i++];
> 
>       if (ctrl < 32) {
>         // Literal run
>         let len = ctrl + 1;
>         if (j + len > outputLimit) len = outputLimit - j;
>         for (let k = 0; k < len; k++) output[j++] = input[i++];
>       } else {
>         // Back-reference
>         let len = ctrl >> 5;
>         let offset = ((ctrl & 0x1f) << 8);
>         if (i < input.length) offset += input[i++];
>         
>         if (len === 7 && i < input.length) {
>           len += input[i++];
>         }
>         len += 2;
> 
>         const start = j - offset - 1;
>         if (start < outputOffset) {
>             // Invalid reference, skip
>             j += len;
>             continue;
>         }
> 
>         for (let k = 0; k < len; k++) {
>           if (j < outputLimit && (start + k) < j) {
>               output[j] = output[start + k];
>               j++;
>           }
>         }
>       }
>     }
>     return [i, j - outputOffset]; // [newInputIdx, writtenBytes]

>         }
>         offset += length;
>     }
